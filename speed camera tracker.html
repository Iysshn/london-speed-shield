    <script>

        // Sample London Camera Locations (Lat, Long, Name)

        const CAMERAS = [

            {lat: 51.5074, lng: -0.1278, name: "Charing Cross"},

            {lat: 51.4988, lng: -0.1749, name: "Cromwell Road"},

            {lat: 51.5201, lng: -0.0102, name: "A12 Blackwall"}

        ];


        function startTracking() {

            if (navigator.geolocation) {

                navigator.geolocation.watchPosition(checkPosition);

                document.querySelector('button').style.display = 'none';

                document.getElementById('status').innerText = "Searching for cameras...";

            }

        }


        function checkPosition(pos) {

            const userLat = pos.coords.latitude;

            const userLng = pos.coords.longitude;

            let closest = 999999;


            CAMERAS.forEach(cam => {

                const dist = calculateDistance(userLat, userLng, cam.lat, cam.lng);

                if (dist < closest) closest = dist;

            });


            document.getElementById('distance').innerText = `Nearest Camera: ${closest.toFixed(0)}m`;


            if (closest < 500) {

                document.body.className = "warning";

                speak("Warning: Speed camera ahead!");

            } else {

                document.body.className = "";

            }

        }


        // Standard distance formula (Haversine)

        function calculateDistance(lat1, lon1, lat2, lon2) {

            const R = 6371e3; 

            const dLat = (lat2-lat1) * Math.PI/180;

            const dLon = (lon2-lon1) * Math.PI/180;

            const a = Math.sin(dLat/2) * Math.sin(dLat/2) +

                      Math.cos(lat1*Math.PI/180) * Math.cos(lat2*Math.PI/180) *

                      Math.sin(dLon/2) * Math.sin(dLon/2);

            return R * 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1-a));

        }


        function speak(text) {

            if (!window.speechSynthesis.speaking) {

                const msg = new SpeechSynthesisUtterance(text);

                window.speechSynthesis.speak(msg);

            }

        }
